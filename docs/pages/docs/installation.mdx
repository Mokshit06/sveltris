import { Tab, Tabs, Steps, Callout } from 'nextra-theme-docs'

# Installation

<Steps>
### Install package
<Tabs items={['pnpm', 'npm', 'yarn']}>
  <Tab>
    ```sh filename="Terminal" copy
    pnpm i sveltris
    ```
  </Tab>
  <Tab>
    ```sh filename="Terminal" copy
    npm i sveltris
    ```
  </Tab>
  <Tab>
    ```sh filename="Terminal" copy
    yarn add sveltris
    ```
  </Tab>
</Tabs>

### Setup plugin

<Tabs items={["Esbuild",  "Vite", "Webpack"]}>
  <Tab>
    ```js
    const { sveltrisEsbuildPlugins } = require("sveltris/esbuild")
    const esbuild = require('esbuild')

    esbuild.build({
      // ... rest of your config
      plugins: [
        ...sveltrisEsbuildPlugins()
      ]
    })
    ```
  </Tab>
  <Tab>
  ```js filename="vite.config.js"
  import { sveltrisVitePlugin } from "sveltris/vite"
  import { defineConfig } from 'vite';

  export default defineConfig({
    plugins: [
      sveltrisVitePlugin()
    ]
  })
  ```
  </Tab>
  <Tab>
  ```js filename="webpack.config.js"
  const { SveltrisWebpackPlugin } = require("sveltris/webpack")

  module.exports = {
    plugins: [
      new SveltrisWebpackPlugin()
    ]
  }
  ```


  </Tab>
</Tabs>
  <Callout>
  You might need to add the bundler plugins for Svelte and React if you haven't already added those.
  </Callout>
</Steps>

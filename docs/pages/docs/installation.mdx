import { Tab, Tabs, Steps, Callout } from 'nextra-theme-docs'

# Installation

<Steps>
### Install package
<Tabs items={['pnpm', 'npm', 'yarn']}>
  <Tab>
    ```sh filename="Terminal" copy
    pnpm i sveltris
    ```
  </Tab>
  <Tab>
    ```sh filename="Terminal" copy
    npm i sveltris
    ```
  </Tab>
  <Tab>
    ```sh filename="Terminal" copy
    yarn add sveltris
    ```
  </Tab>
</Tabs>

### Setup plugin

<Tabs items={["Esbuild",  "Vite", "Webpack"]}>
  <Tab>
    ```js
    const { sveltrisEsbuildPlugins } = require("sveltris")
    const esbuild = require('esbuild')

    esbuild.build({
      // ... rest of your config
      plugins: [
        ...sveltrisEsbuildPlugins()
      ]
    })
    ```
  </Tab>
  <Tab>
  ```js filename="vite.config.js"
  import { sveltrisVitePlugins } from "sveltris"
  import { defineConfig } from 'vite';

  export default defineConfig({
    plugins: [
      ...sveltrisVitePlugins()
    ]
  })
  ```
  </Tab>
  <Tab>
  ```js filename="webpack.config.js"
  const { sveltrisWebpackPlugins } = require("sveltris")

  module.exports = {
    plugins: [
      ...sveltrisWebpackPlugins()
    ]
  }
  ```
  </Tab>
    <Tab>
  ```js filename="rollup.config.js"
  const { sveltrisRollupPlugins } = require("sveltris")

  module.exports = {
    plugins: [
      ...sveltrisRollupPlugins()
    ]
  }
  ```
  </Tab>
</Tabs>
  <Callout>
  You might need to add the bundler plugins for Svelte and React if you haven't already added those.
  </Callout>
</Steps>
